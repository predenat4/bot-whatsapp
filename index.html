Bot WhatsApp - Guide Complet

Avec commandes de groupe (/kickall, /tagall, /clear, etc.)

📦 Installation

Étape 1 : Node.js

Télécharger Node.js (inclut npm) depuis nodejs.org

Étape 2 : Créer le dossier

mkdir my-whatsapp-bot
cd my-whatsapp-bot

Étape 3 : Initialiser npm

npm init -y

Étape 4 : Installer les dépendances

npm install whatsapp-web.js qrcode-terminal dotenv fs

✅ Prêt à continuer !

Copiez les fichiers dans l'onglet Fichiers

const { Client, LocalAuth, MessageMedia } = require('whatsapp-web.js');
const qrcode = require('qrcode-terminal');
const fs = require('fs');
require('dotenv').config();

const client = new Client({
  authStrategy: new LocalAuth()
});

// Stockage des données
const botData = {
  bannedUsers: {},
  mutedUsers: {},
  userWarnings: {},
  groupSettings: {},
  logs: []
};

// Fichier de sauvegarde
const dataFile = 'botData.json';
if (fs.existsSync(dataFile)) {
  const saved = JSON.parse(fs.readFileSync(dataFile, 'utf8'));
  Object.assign(botData, saved);
}

function saveData() {
  fs.writeFileSync(dataFile, JSON.stringify(botData, null, 2));
}

// Commandes disponibles
const COMMANDS = {
  '/help': 'Affiche toutes les commandes',
  '/tagall': 'Tagger tous les membres du groupe',
  '/kickall': 'Expulser tous les membres',
  '/kick @user': 'Expulser un utilisateur spécifique',
  '/clear': 'Supprimer les messages',
  '/info': 'Infos sur le groupe',
  '/promote @user': 'Promouvoir admin',
  '/demote @user': 'Rétrograder admin',
  '/ban @user': 'Bannir un utilisateur',
  '/unban @user': 'Débannir',
  '/warn @user': 'Avertir',
  '/mute @user': 'Rendre muet',
  '/unmute @user': 'Retirer le silence',
  '/profile @user': 'Voir le profil',
  '/sudo @user': 'Mode délégué',
  '/logs': 'Voir les logs',
  '/stats': 'Statistiques',
  '/leave': 'Bot quitte le groupe'
};

function addLog(groupId, action, user) {
  botData.logs.push({
    timestamp: new Date().toISOString(),
    group: groupId,
    action,
    user
  });
  if (botData.logs.length > 100) botData.logs.shift();
  saveData();
}

client.on('qr', qr => {
  console.log('Scannez ce QR code avec WhatsApp:');
  qrcode.generate(qr, { small: true });
});

client.on('ready', () => {
  console.log('✅ Bot connecté avec succès !');
});

client.on('message', async msg => {
  try {
    const chat = await msg.getChat();
    
    if (!chat.isGroup) return;

    const contact = await msg.getContact();
    const isAdmin = msg.author === process.env.ADMIN_ID;
    const isMuted = botData.mutedUsers[msg.from]?.[msg.author];
    
    if (!msg.body.startsWith('/')) return;
    
    // Vérifier si l'utilisateur est muet
    if (isMuted) {
      await msg.react('🔇');
      return;
    }

    const args = msg.body.slice(1).split(' ');
    const command = '/' + args[0];
    const target = args[1];

    console.log(`[${chat.name}] ${contact.name}: ${msg.body}`);

    switch (command) {
      case '/help':
        let helpText = '📋 *Commandes disponibles:*\n\n';
        Object.entries(COMMANDS).forEach(([cmd, desc]) => {
          helpText += `${cmd} - ${desc}\n`;
        });
        await msg.reply(helpText);
        break;

      case '/tagall':
        let tagMessage = '👋 *Ping à tous:*\n';
        const members = await chat.participants;
        members.forEach(m => {
          tagMessage += `@${m.id.user} `;
        });
        await client.sendMessage(msg.from, tagMessage, { mentions: members });
        addLog(msg.from, 'tagall', contact.name);
        break;

      case '/kickall':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        const allMembers = await chat.participants;
        for (let member of allMembers) {
          if (member.id.user !== process.env.ADMIN_ID.split('@')[0]) {
            await chat.removeParticipants([member.id]);
            await new Promise(resolve => setTimeout(resolve, 500));
          }
        }
        await msg.reply('✅ Tous les membres expulsés!');
        addLog(msg.from, 'kickall', contact.name);
        break;

      case '/kick':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message de l\'utilisateur');
          break;
        }
        const quotedMsg = await msg.getQuotedMessage();
        const targetContact = await quotedMsg.getContact();
        const targetMember = await chat.participants.find(p => p.id.user === quotedMsg.author.split('@')[0]);
        
        if (targetMember) {
          await chat.removeParticipants([targetMember.id]);
          await msg.reply(`✅ @${targetContact.name} a été expulsé!`);
          addLog(msg.from, 'kick', targetContact.name);
        }
        break;

      case '/sudo':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message de l\'utilisateur');
          break;
        }
        const sudoTarget = await msg.getQuotedMessage();
        const sudoUser = await sudoTarget.getContact();
        
        // Enregistrer cet utilisateur comme admin temporaire
        if (!botData.groupSettings[msg.from]) {
          botData.groupSettings[msg.from] = {};
        }
        botData.groupSettings[msg.from].sudoUser = sudoTarget.author;
        saveData();
        
        await msg.reply(`✅ @${sudoUser.name} peut maintenant utiliser les commandes admin!`);
        addLog(msg.from, 'sudo', sudoUser.name);
        break;

      case '/ban':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message');
          break;
        }
        const bannedMsg = await msg.getQuotedMessage();
        const bannedUser = await bannedMsg.getContact();
        
        if (!botData.bannedUsers[msg.from]) {
          botData.bannedUsers[msg.from] = {};
        }
        botData.bannedUsers[msg.from][bannedMsg.author] = true;
        saveData();
        
        await msg.reply(`🚫 @${bannedUser.name} a été banni!`);
        addLog(msg.from, 'ban', bannedUser.name);
        break;

      case '/unban':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message');
          break;
        }
        const unbannedMsg = await msg.getQuotedMessage();
        const unbannedUser = await unbannedMsg.getContact();
        
        if (botData.bannedUsers[msg.from]) {
          delete botData.bannedUsers[msg.from][unbannedMsg.author];
          saveData();
        }
        
        await msg.reply(`✅ @${unbannedUser.name} a été débanni!`);
        addLog(msg.from, 'unban', unbannedUser.name);
        break;

      case '/warn':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message');
          break;
        }
        const warnedMsg = await msg.getQuotedMessage();
        const warnedUser = await warnedMsg.getContact();
        
        if (!botData.userWarnings[msg.from]) {
          botData.userWarnings[msg.from] = {};
        }
        if (!botData.userWarnings[msg.from][warnedMsg.author]) {
          botData.userWarnings[msg.from][warnedMsg.author] = 0;
        }
        botData.userWarnings[msg.from][warnedMsg.author]++;
        const warnings = botData.userWarnings[msg.from][warnedMsg.author];
        saveData();
        
        await msg.reply(`⚠️ Avertissement à @${warnedUser.name} (${warnings}/3)`);
        
        if (warnings >= 3) {
          const warnMember = await chat.participants.find(p => p.id.user === warnedMsg.author.split('@')[0]);
          if (warnMember) {
            await chat.removeParticipants([warnMember.id]);
            await msg.reply(`🚫 @${warnedUser.name} a été expulsé (3 avertissements)`);
            delete botData.userWarnings[msg.from][warnedMsg.author];
            saveData();
          }
        }
        addLog(msg.from, 'warn', warnedUser.name);
        break;

      case '/mute':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message');
          break;
        }
        const mutedMsg = await msg.getQuotedMessage();
        const mutedUser = await mutedMsg.getContact();
        
        if (!botData.mutedUsers[msg.from]) {
          botData.mutedUsers[msg.from] = {};
        }
        botData.mutedUsers[msg.from][mutedMsg.author] = true;
        saveData();
        
        await msg.reply(`🔇 @${mutedUser.name} ne peut plus parler!`);
        addLog(msg.from, 'mute', mutedUser.name);
        break;

      case '/unmute':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message');
          break;
        }
        const unmutedMsg = await msg.getQuotedMessage();
        const unmutedUser = await unmutedMsg.getContact();
        
        if (botData.mutedUsers[msg.from]) {
          delete botData.mutedUsers[msg.from][unmutedMsg.author];
          saveData();
        }
        
        await msg.reply(`🔊 @${unmutedUser.name} peut à nouveau parler!`);
        addLog(msg.from, 'unmute', unmutedUser.name);
        break;

      case '/profile':
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message');
          break;
        }
        const profileMsg = await msg.getQuotedMessage();
        const profileUser = await profileMsg.getContact();
        const warnings = botData.userWarnings[msg.from]?.[profileMsg.author] || 0;
        const isBanned = botData.bannedUsers[msg.from]?.[profileMsg.author] || false;
        const isMutedProfile = botData.mutedUsers[msg.from]?.[profileMsg.author] || false;
        
        let profileText = `👤 *Profil de @${profileUser.name}*\n`;
        profileText += `⚠️ Avertissements: ${warnings}/3\n`;
        profileText += `🚫 Banni: ${isBanned ? 'Oui' : 'Non'}\n`;
        profileText += `🔇 Muet: ${isMutedProfile ? 'Oui' : 'Non'}`;
        
        await msg.reply(profileText);
        break;

      case '/info':
        const groupInfo = `
ℹ️ *Informations du groupe*
Nom: ${chat.name}
Membres: ${chat.participants.length}
Description: ${chat.description || 'Aucune'}
        `;
        await msg.reply(groupInfo);
        break;

      case '/stats':
        const recentLogs = botData.logs.filter(l => l.group === msg.from).slice(-10);
        let statsText = '📊 *Statistiques*\n';
        statsText += `Membres: ${chat.participants.length}\n`;
        statsText += `Bannies: ${Object.keys(botData.bannedUsers[msg.from] || {}).length}\n`;
        statsText += `Muets: ${Object.keys(botData.mutedUsers[msg.from] || {}).length}\n`;
        statsText += `\n*Dernières actions:*\n`;
        recentLogs.forEach(log => {
          statsText += `${log.action} - ${log.user}\n`;
        });
        await msg.reply(statsText);
        break;

      case '/logs':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        const logs = botData.logs.filter(l => l.group === msg.from).slice(-20);
        let logsText = '📋 *Derniers logs:*\n\n';
        logs.forEach(log => {
          logsText += `${new Date(log.timestamp).toLocaleString()} - ${log.action} (${log.user})\n`;
        });
        await msg.reply(logsText || '✅ Aucun log');
        break;

      case '/clear':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        const allMessages = await chat.fetchMessages({ limit: 1000 });
        await Promise.all(allMessages.map(m => m.delete(true)));
        await msg.reply('✅ Messages supprimés!');
        addLog(msg.from, 'clear', contact.name);
        break;

      case '/promote':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message');
          break;
        }
        const promoteMsg = await msg.getQuotedMessage();
        const promotedUser = await promoteMsg.getContact();
        const promoteMember = await chat.participants.find(p => p.id.user === promoteMsg.author.split('@')[0]);
        
        if (promoteMember) {
          await chat.promoteParticipants([promoteMember.id]);
          await msg.reply(`✅ @${promotedUser.name} est admin!`);
          addLog(msg.from, 'promote', promotedUser.name);
        }
        break;

      case '/demote':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        if (!msg.hasQuotedMsg) {
          await msg.reply('❌ Répondez au message');
          break;
        }
        const demoteMsg = await msg.getQuotedMessage();
        const demotedUser = await demoteMsg.getContact();
        const demoteMember = await chat.participants.find(p => p.id.user === demoteMsg.author.split('@')[0]);
        
        if (demoteMember) {
          await chat.demoteParticipants([demoteMember.id]);
          await msg.reply(`✅ @${demotedUser.name} n'est plus admin`);
          addLog(msg.from, 'demote', demotedUser.name);
        }
        break;

      case '/leave':
        if (!isAdmin) {
          await msg.reply('❌ Non autorisé!');
          break;
        }
        await chat.leave();
        break;

      default:
        await msg.reply('❓ Commande inconnue. Tapez */help*');
    }
  } catch (err) {
    console.error('Erreur:', err);
  }
});

🚀 Déploiement

Local (Recommandé pour commencer)

  1. Ouvrez un terminal dans votre dossier
  2. Tapez: npm start
  3. Scannez le QR code avec WhatsApp
  4. Le bot est maintenant actif ✅

Hébergement Cloud (Production)

Options populaires:

  • ✅ Render.com - Gratuit avec limite
  • ✅ Railway.app - $5/mois
  • ✅ Heroku - À partir de $7/mois
  • ✅ AWS EC2 - Couche gratuite 12 mois
  • ✅ DigitalOcean - $4/mois

Note: Le bot doit rester actif 24/7 pour fonctionner

Etapes de déploiement

  1. Créer un compte sur la plateforme (Render, Railway, etc.)
  2. Connecter votre repo Git (ou uploadez les fichiers)
  3. Définir l'environnement: npm start
  4. Configurer les variables d'environnement (.env)
  5. Déployer et garder actif 24/7

⚠️ Important

  • 🔒 Ne partagez JAMAIS votre fichier .env
  • 🔑 Gardez vos credentials secrets
  • 📱 Le bot doit rester connecté en permanence
  • ⚡ Certaines commandes peuvent être limitées par WhatsApp
